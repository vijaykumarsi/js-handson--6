Q.1. Find the number of digits 
            const Find_Digits = (N) => 
{
  let ans =0;
  while(N!==0){
    ans++;
    N=Math.floor(N/10);
  }
  return ans;
};

Q.3. Find Sum. 
                const findSum = (n) => {
                    let sum=0

                    }
