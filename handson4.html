Q.10. Min and Max. 
                function arrayMin(arr) {
                    let min=arr[0]
                    for(let i=0;i < arr.length;++i){
                      if(arr[i] < min){
                        min=arr[i]
                      }
                       
                    }
                    return min
                    
                    
                    }
                    
                    function arrayMax(arr) {
                     max=arr[0]
                    for(let i=0;i < arr.length;++i){

                        max=arr[i]
                      }
                       
                    }

Q.11. Birthday Game . 
                
                function Birthday_Game(arr,D ,M) {
 
                    let N=arr.length;
                    let count=0;
                    for(let i=0;i < N-M+1;i++){
                      let sum=0;
                      for(let j=i;j < M+i;j++){
                        sum+=arr[j];
                      }
                      if(sum===D)
                      count++;
                    }
                    return count;
                    }
