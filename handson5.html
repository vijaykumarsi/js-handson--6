Q.2.Count the Heads 
                var Count_Occ = (s) => 
                {
                  let n=s.length;
                  let str=s.split("").sort().join("");
                  let ans ="";
                  for (let i=0;i < n;i++){
                    let count=0;
                    for(let j=0;j < n;j++){
                      if(str[i]===str[j]&& i > j){
                        break;
                      }
                      if (str[i]===str[j]){
                        count+=1;
                      }
                    }
                    if (count>1){
                      ans+=str[i]+count;
                    }
                  }
                  return ans;
                };

                {

                  let countd=0;

                    if(S.charAt(i)==='A'){
                      counta++;
                    }
                    else if (S.charAt(i)==='D'){
                      countd++;
                    }
                    
                  }
                  if(counta>countd){
                    return 'Aditya';
                  }
                  else if (counta< countd){

                }
