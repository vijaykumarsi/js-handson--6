Q.1. Find the Smaller Angle 
            var Minimal_Angle = (h, m) => 
            {
              h=h%12;
              let hour =(h*360)/12 + (m*360)/(12*60);
              let min = (m*360)/60;
              let angle =Math.abs(hour-min);
              
              
              if (angle>180){
                return 360-angle;
                
              }else{
                return angle;
              }
            };
Q.2.Check whether the year is Leap year or not. 
            var Check_Leap = (year) => 
            {
              if (year % 400===0 && year%100 ===0){
                return 'Leap Year'
              }
              else if (year%4===0 && year%100 !==0){
                
                return 'Leap Year'
              }
              else{
                return "Non Leap Year"
              }
             
              
            
            };
Q.3. Perfect Number Check. 
            var Perfect_Check = (N) => 
            {
               let ans=[];
               let j=0;
               for(let i=1; i < N; i++){
                 if(N%i===0)
                 ans[j]=ans[j]+i;
                 
               }
               if(ans[j]===N)
                  return "YES";
                 else
                 return "NO";
                 
                 
                    
            };
Q.4.Reverse a Number. 
            var Reverse_Number = (N) => 
            {
              let ans = Number(String(N).split('').reverse().join(''));
              return ans;
            };
